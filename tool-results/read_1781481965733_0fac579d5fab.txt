   279→  const [response, setResponse] = useState('');
   280→  const [analysis, setAnalysis] = useState<Array<{ label: string; severity: string; detail: string }>>([]);
   281→  const [isLoading, setIsLoading] = useState(false);
   282→  const [activePreset, setActivePreset] = useState<number | null>(null);
   283→
   284→  const analyzePayload = useCallback((prompt: string) => {
   285→    if (!config) return [];
   286→    const results: Array<{ label: string; severity: string; detail: string }> = [];
   287→    for (const { pattern, label, severity } of config.analysisPatterns) {
   288→      if (pattern.test(prompt)) {
   289→        results.push({ label, severity, detail: `Обнаружен паттерн: ${label}` });
   290→      }
   291→    }
   292→    return results;
   293→  }, [config]);
   294→
   295→  const handleSubmit = useCallback(async () => {
   296→    if (!userPrompt.trim()) return;
   297→    setIsLoading(true);
   298→    setAnalysis(analyzePayload(userPrompt));
   299→
   300→    try {
   301→      const res = await fetch('/api/chat', {
   302→        method: 'POST',
   303→        headers: { 'Content-Type': 'application/json' },
   304→        body: JSON.stringify({
   305→          messages: [
   306→            { role: 'system', content: systemPrompt },
   307→            { role: 'user', content: userPrompt },
   308→          ],
   309→        }),
   310→      });
   311→      const data = await res.json();
   312→      setResponse(data.content || data.error || 'Нет ответа');
   313→    } catch {
   314→      setResponse('Ошибка: не удалось подключиться к API. Добавьте API-ключ в настройках модели.');
   315→    } finally {
   316→      setIsLoading(false);
   317→    }
   318→  }, [userPrompt, systemPrompt, analyzePayload]);
   319→
   320→  const handleReset = useCallback(() => {
   321→    setSystemPrompt(defaultSystem || '');
   322→    setUserPrompt(defaultPrompt || '');
   323→    setResponse('');
   324→    setAnalysis([]);
   325→    setActivePreset(null);
   326→  }, [defaultPrompt, defaultSystem]);
   327→
   328→  const handlePreset = useCallback((idx: number) => {
   329→    if (!config) return;
   330→    const preset = config.presets[idx];
   331→    setUserPrompt(preset.prompt);
   332→    if (preset.system) setSystemPrompt(preset.system);
   333→    setActivePreset(idx);
   334→    setResponse('');
   335→    setAnalysis([]);
   336→  }, [config]);
   337→
   338→  const severityColor = (s: string) => {
   339→    switch (s) {
   340→      case 'critical': return 'bg-red-500/10 text-red-500 border-red-500/20';
   341→      case 'high': return 'bg-orange-500/10 text-orange-500 border-orange-500/20';
   342→      case 'medium': return 'bg-amber-500/10 text-amber-500 border-amber-500/20';
   343→      default: return 'bg-blue-500/10 text-blue-500 border-blue-500/20';
   344→    }
   345→  };
   346→
   347→  return (
   348→    <Card className="border-red-500/20">
   349→      <CardHeader className="pb-3">
   350→        <div className="flex items-center gap-2">
   351→          <div className="p-1.5 rounded-lg bg-red-500/10">
   352→            <FlaskConical className="h-4 w-4 text-red-500" />
   353→          </div>
   354→          <CardTitle className="text-lg">{title}</CardTitle>
   355→          <Badge variant="outline" className="ml-auto text-red-500 border-red-500/30 text-xs">
   356→            RED TEAM
   357→          </Badge>
   358→        </div>
   359→        <p className="text-sm text-muted-foreground">{description}</p>
   360→      </CardHeader>
   361→      <CardContent className="space-y-4">
   362→        {/* Presets */}
   363→        {config && config.presets.length > 0 && (
   364→          <div>
   365→            <p className="text-sm font-medium mb-2 flex items-center gap-1.5">
   366→              <Target className="h-3.5 w-3.5 text-red-500" />
   367→              Готовые пейлоады:
   368→            </p>
   369→            <div className="flex flex-wrap gap-2">
   370→              {config.presets.map((preset, i) => (
   371→                <Button
   372→                  key={i}
   373→                  variant={activePreset === i ? 'default' : 'outline'}
   374→                  size="sm"
   375→                  onClick={() => handlePreset(i)}
   376→                  className={activePreset === i ? 'bg-red-500 hover:bg-red-600' : ''}
   377→                >
   378→                  {preset.label}
   379→                </Button>
   380→              ))}
   381→            </div>
   382→          </div>
   383→        )}
   384→
   385→        <Separator />
   386→
   387→        {/* System Prompt */}
   388→        <div>
   389→          <p className="text-sm font-medium mb-1.5 flex items-center gap-1.5">
   390→            <Shield className="h-3.5 w-3.5 text-emerald-500" />
   391→            Системный промпт (защита):
   392→          </p>
   393→          <Textarea
   394→            value={systemPrompt}
   395→            onChange={e => setSystemPrompt(e.target.value)}
   396→            placeholder="Системный промпт модели..."
   397→            className="min-h-[60px] text-sm font-mono"
   398→          />
   399→        </div>
   400→
   401→        {/* User Prompt */}
   402→        <div>
   403→          <p className="text-sm font-medium mb-1.5 flex items-center gap-1.5">
   404→            <Skull className="h-3.5 w-3.5 text-red-500" />
   405→            Атакующий промпт:
   406→          </p>
   407→          <Textarea
   408→            value={userPrompt}
   409→            onChange={e => setUserPrompt(e.target.value)}
   410→            placeholder="Введите вредоносный промпт..."
   411→            className="min-h-[80px] text-sm font-mono"
   412→          />
   413→        </div>
   414→
   415→        {/* Analysis */}
   416→        {analysis.length > 0 && (
   417→          <div className={`p-3 rounded-lg border ${dangerBg}`}>
   418→            <p className="text-sm font-medium mb-2 flex items-center gap-1.5 text-red-500">
   419→              <ShieldAlert className="h-4 w-4" />
   420→              Анализ пейлоада:
   421→            </p>
   422→            <div className="space-y-1.5">
   423→              {analysis.map((a, i) => (
   424→                <div key={i} className={`px-2.5 py-1 rounded text-xs font-medium border ${severityColor(a.severity)}`}>
   425→                  [{a.severity.toUpperCase()}] {a.label}
   426→                </div>
   427→              ))}
   428→            </div>
   429→          </div>
   430→        )}
   431→
   432→        {/* Buttons */}
   433→        <div className="flex gap-2">
   434→          <Button onClick={handleSubmit} disabled={isLoading} className="bg-red-500 hover:bg-red-600 text-white">
   435→            {isLoading ? (
   436→              <span className="flex items-center gap-1.5">
   437→                <span className="animate-spin">⚡</span> Атака...
   438→              </span>
   439→            ) : (
   440→              <span className="flex items-center gap-1.5">
   441→                <Play className="h-3.5 w-3.5" /> Запустить атаку
   442→              </span>
   443→            )}
   444→          </Button>
   445→          <Button variant="outline" onClick={handleReset} size="sm">
   446→            <RotateCcw className="h-3.5 w-3.5 mr-1" /> Сброс
   447→          </Button>
   448→        </div>
   449→
   450→        {/* Response */}
   451→        {response && (
   452→          <div className={`p-4 rounded-lg border ${analysis.length > 0 ? dangerBg : successBg}`}>
   453→            <p className="text-sm font-medium mb-2 flex items-center gap-1.5">
   454→              {analysis.length > 0 ? (
   455→                <><ShieldAlert className="h-4 w-4 text-red-500" /> Ответ модели (атака может быть успешна):</>
   456→              ) : (
   457→                <><ShieldCheck className="h-4 w-4 text-emerald-500" /> Ответ модели (защита устояла):</>
   458→              )}
   459→            </p>
   460→            <pre className="text-sm whitespace-pre-wrap font-mono bg-background/50 p-3 rounded-lg border border-border">
   461→              {response}
   462→            </pre>
   463→          </div>
   464→        )}
   465→
   466→        {/* Tips */}
   467→        {config && config.tips.length > 0 && (
   468→          <div className={`p-3 rounded-lg border ${infoBg}`}>
   469→            <p className="text-sm font-medium mb-1.5 flex items-center gap-1.5">
   470→              <Zap className="h-3.5 w-3.5 text-red-500" />
   471→              Советы для Red Team:
   472→            </p>
   473→            <ul className="space-y-1">
   474→              {config.tips.map((tip, i) => (
   475→                <li key={i} className="text-xs text-muted-foreground flex items-start gap-1.5">
   476→                  <span className="text-red-500 mt-0.5">▸</span>
   477→                  {tip}
   478→                </li>
   479→              ))}
   480→            </ul>
   481→          </div>
   482→        )}
   483→      </CardContent>
   484→    </Card>
   485→  );
   486→}
